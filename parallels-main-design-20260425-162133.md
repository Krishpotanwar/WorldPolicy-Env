## Problem Statement

WorldPolicy-Env V6.1 is a working multi-agent geopolitical debate simulator — 7 AI agents
(USA, China, Russia, India, DPRK, Saudi Arabia, UNESCO) debate live crises, vote on policy
actions, and update a Country P&L ledger. The demo quality is high. But the submission is
currently invalid for the hackathon: no OpenEnv compliance, no reward system, no training
script. Additionally, the debate visualization is disconnected from the globe (no speaking
country highlight), and the P&L/relationship data is pre-loaded rather than live.

This design upgrades V6.1 into a fully OpenEnv-compliant RL training environment while
preserving and extending the visual demo quality.

---

## What Makes This Cool

1. **The only geopolitical RL env with live real-world data**: Crisis events come from GDELT
   (every 15 min). Country P&L baselines come from World Bank. Personas update dynamically
   from recent events. Nobody else at this hackathon is doing this.

2. **The globe pulses as agents speak**: When India speaks, India's dot glows. When a
   coalition forms, alliance arcs animate. This is the visual anchor judges remember.

3. **Composable reward rubric**: consensus_quality + coalition_coherence + resolution_speed +
   UNESCO_mandate_compliance = a reward function with real research value. Not just +1/-1.

4. **Dynamic personas**: Before each debate, the orchestrator fetches the last 24h of GDELT
   headlines for each country and injects them into the persona prompt. North Korea's rhetoric
   shifts when there's actually a nuclear test in the news.

#### New file: `pytorch_scorer.py` — PyTorch requirement (trains in < 30s)
```python
import torch
import torch.nn as nn
import json
from pathlib import Path

class StabilityScorer(nn.Module):
    """
    6-layer MLP: estimates world stability score from economic + diplomatic features.
    Input: 12 features (2 per country: gdp_norm, relationship_avg)
    Output: scalar stability score in [0, 1]
    Trains on synthetic episodes in ~20 seconds on CPU.
    """
    def __init__(self):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(12, 32), nn.ReLU(),
            nn.Linear(32, 16), nn.ReLU(),
            nn.Linear(16, 8),  nn.ReLU(),
            nn.Linear(8, 4),   nn.ReLU(),
            nn.Linear(4, 2),   nn.ReLU(),
            nn.Linear(2, 1),   nn.Sigmoid(),
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.net(x)

    @classmethod
    def features_from_state(cls, country_pnl: dict, rel_matrix: dict) -> torch.Tensor:
        AGENTS = ["USA", "CHN", "RUS", "IND", "DPRK", "SAU"]
        GDP_MAX = 30e12
        feats = []
        for aid in AGENTS:
            gdp      = country_pnl.get(aid, {}).get("gdp", 1e12)
            rel_vals = list(rel_matrix.get(aid, {}).values())
            rel_avg  = sum(rel_vals) / max(len(rel_vals), 1)
            feats.extend([gdp / GDP_MAX, (rel_avg + 1.0) / 2.0])  # normalize to [0,1]
        return torch.tensor(feats, dtype=torch.float32).unsqueeze(0)


def train_scorer(weights_path: str = "scorer_weights.pt", n_samples: int = 50_000):
    """Generate synthetic training data and train in < 30 seconds on CPU."""
    import random
    model = StabilityScorer()
    optimizer = torch.optim.Adam(model.parameters(), lr=1e-3)
    loss_fn = nn.MSELoss()

    for _ in range(n_samples // 100):
        # Synthetic batch: random features, label = mean of all features
        x = torch.rand(100, 12)
        y = x.mean(dim=1, keepdim=True)
        pred = model(x)
        loss = loss_fn(pred, y)
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()

    torch.save(model.state_dict(), weights_path)
    print(f"StabilityScorer trained + saved to {weights_path}")


_scorer: StabilityScorer | None = None

def score_stability(country_pnl: dict, rel_matrix: dict) -> float:
    """Load scorer (cached) and return stability estimate."""
    global _scorer
    if _scorer is None:
        _scorer = StabilityScorer()
        weights = Path("scorer_weights.pt")
        if weights.exists():
            _scorer.load_state_dict(torch.load(weights, map_location="cpu"))
        _scorer.eval()
    x = StabilityScorer.features_from_state(country_pnl, rel_matrix)
    with torch.no_grad():
        return float(_scorer(x).item())


if __name__ == "__main__":
    train_scorer()
```

async def grader(body: dict):
    """
    Score a completed episode across all grader dimensions.
    Returns per-dimension scores + composite, all in [0.0, 1.0].
    """
    session_id  = body.get("session_id")
    task        = body.get("task", "task_1")
    rounds      = body.get("rounds", [])      # list of round_result dicts
    
    grader_cls  = TASK_GRADERS.get(task, CrisisResolutionGrader)()
    episode_score = sum(grader_cls.score(r) for r in rounds) / max(len(rounds), 1)
    
    # Normalize to [0, 1] using DisasterMan's formula
    import math
    max_steps   = TASKS.get(task, {}).get("max_steps", 5)
    normalized  = (math.tanh(episode_score / max_steps * 2) + 1) / 2
    
    return {
        "task":           task,
        "session_id":     session_id,
        "raw_score":      round(episode_score, 4),
        "normalized":     round(normalized, 4),
        "step_count":     len(rounds),
        "target_range":   TASKS.get(task, {}).get("target_reward_range", (0.4, 0.8)),
    }

# WorldPolicyEnvironment — OpenEnv-compliant wrapper around DebateOrchestrator
# NOTE: Replace `openenv.Environment` with the actual base class name from the SDK
from openenv import Environment  # verify import path before writing

class WorldPolicyEnvironment(Environment):
    """
    OpenEnv-compliant geopolitical RL env.
    Session-isolated: each GRPO rollout gets its own session_id → independent world state.
    """

    def __init__(self):
        self._sessions: dict = {}    # session_id → WorldState
        self._orchestrator = DebateOrchestrator()
        self._persona_loader = PersonaLoader()
        self._graders = {
            "crisis_resolution":    CrisisResolutionGrader(),
            "coalition_formation":  CoalitionGrader(),
            "diplomatic_negotiation": DiplomacyGrader(),
        }

    def reset(self, task: str = "crisis_resolution", session_id: str = None) -> dict:
        """
        Initialize a new episode.
        Returns: initial observation dict (matches observation_space in openenv.yaml).
        """
        import uuid
        sid = session_id or str(uuid.uuid4())
        
        # Live crisis from GDELT, fallback to static
        crisis = get_live_crisis(task_to_crisis_type(task))
        
        # P&L baselines from World Bank, fallback to WB_FALLBACKS
        pnl = {aid: get_wb_baseline(aid) for aid in AGENT_IDS if aid != "UNESCO"}
        
        # Relationship matrix: load from relationships.json, will update live
        rel_matrix = self._persona_loader.load_relationships()
        
        self._sessions[sid] = {
            "task": task,
            "step": 0,
            "max_steps": 5,
            "crisis": crisis,
            "country_pnl": pnl,
            "relationship_matrix": rel_matrix,
            "debate_history": [],
            "prev_stability": self._stability_score(pnl, rel_matrix),
            "active_agent": "USA",
            "done": False,
        }
        return {"session_id": sid, "observation": self._observation(sid)}

    def step(self, action: dict, session_id: str = None) -> tuple[dict, float, bool, dict]:
        """
        Execute one policy action.
        action_type: propose_resolution | form_coalition | veto | abstain |
                     invoke_article | sanction
        Returns: (observation, reward, done, info)
        """
        sid = session_id or list(self._sessions.keys())[-1]
        state = self._sessions[sid]
        
        if state["done"]:
            return self._observation(sid), 0.0, True, {"error": "episode already done"}
        
        # Run debate round with this action as the MAPPO proposal
        round_result = self._orchestrator.run_round(
            action=action,
            crisis=state["crisis"],
            world_state=state,
            live_events={aid: get_country_events(aid) for aid in AGENT_IDS},
        )
        
        # Compute null-action baseline for counterfactual advantage
        null_round = self._orchestrator.run_null_round(state["crisis"], state)
        round_result["null_action_stability"] = self._stability_score(
            null_round["pnl_after"], null_round["rel_after"]
        )
        round_result["current_stability"] = self._stability_score(
            round_result["pnl_after"], round_result["rel_after"]
        )
        round_result["prev_stability"]       = state["prev_stability"]
        round_result["relationship_matrix"]  = state["relationship_matrix"]
        round_result["next_relationship_matrix"] = round_result["rel_after"]
        round_result["crisis_type"]          = state["crisis"].get("type", "DEFAULT")
        round_result["crisis_resolved"]      = round_result.get("vote_passed", False)
        
        # Apply reward
        grader = self._graders.get(state["task"], self._graders["crisis_resolution"])
        reward = grader.score(round_result)
        
        # Update state
        state["country_pnl"]          = round_result["pnl_after"]
        state["relationship_matrix"]  = round_result["rel_after"]
        state["debate_history"].extend(round_result.get("utterances", []))
        state["prev_stability"]       = round_result["current_stability"]
        state["step"]                += 1
        
        done = state["step"] >= state["max_steps"] or round_result.get("crisis_resolved") or \
               "nuclear_escalation" in round_result.get("constraint_violations", [])
        state["done"] = done
        
        return self._observation(sid), reward, done, {"round": round_result}

    def state(self, session_id: str = None) -> dict:
        sid = session_id or list(self._sessions.keys())[-1]
        return self._sessions.get(sid, {})

    def _observation(self, sid: str) -> dict:
        s = self._sessions[sid]
        return {
            "country_pnl":         s["country_pnl"],
            "relationship_matrix": s["relationship_matrix"],
            "current_crisis":      s["crisis"],
            "debate_history":      s["debate_history"][-6:],  # last 6 utterances
            "active_agent":        s["active_agent"],
            "step_count":          s["step"],
        }

    def _stability_score(self, pnl: dict, rel_matrix: dict) -> float:
        """Heuristic world stability — used for long-horizon V(s) and counterfactual baseline."""
        if not rel_matrix:
            return 0.5
        all_vals = [v for row in rel_matrix.values() for v in row.values()]
        avg_trust = sum(all_vals) / max(len(all_vals), 1)
        return max(0.0, min(1.0, (avg_trust + 1.0) / 2.0))
```

#### New file: `graders.py` — MOGSR Architecture

**MOGSR — Multi-Objective Geopolitical Stability Reward**

This is a 4-layer reward stack designed for research credibility. The pitch:
*"Our agents are not rewarded for passing resolutions; they are rewarded for improving
global system stability under multi-objective geopolitical constraints."*

```
R_final = R_immediate + γ·V(s_{t+1}) + λ·A_counterfactual + β·R_robust
```

    "illegal_aggression":   -0.5,
    "contradictory_policy": -0.2,   # agent proposes opposite of prior stated stance
    "coalition_collapse":   -0.3,   # formed coalition broke apart in the same round
    "un_charter_violation": -0.4,   # UNESCO flagged action as mandate breach
}

GAMMA = 0.95   # discount for long-horizon value
LAMBDA = 0.30  # counterfactual advantage weight
BETA   = 0.10  # shock-robustness weight


def _security_score(r: dict) -> float:
    """S: penalize escalation, reward de-escalation and ceasefire signals."""
    delta_conflict   = r.get("conflict_delta", 0.0)        # >0 = more conflict
    pr_escalation    = r.get("pr_escalation", 0.0)         # [0,1]
    pr_ceasefire     = r.get("pr_ceasefire", 0.0)          # [0,1]
    spillover_risk   = r.get("spillover_risk", 0.0)        # [0,1] contagion
    raw = -delta_conflict - pr_escalation + pr_ceasefire - spillover_risk * 0.5
    return max(0.0, min(1.0, (raw + 2.0) / 4.0))  # normalize to [0,1]


def _diplomacy_score(r: dict) -> float:
    """D: efficiency of resolution — fast peaceful outcomes rewarded most."""
    success       = float(r.get("resolution_success", r.get("vote_passed", False)))
    steps_taken   = max(1, r.get("negotiation_steps", 1))
    max_steps     = r.get("max_negotiation_steps", 5)
    efficiency    = success * (1.0 - (steps_taken - 1) / max_steps)  # penalize dragging on
    return max(0.0, min(1.0, efficiency))


def _coalition_score(r: dict) -> float:
    """C: DYNAMIC coherence from live relationship matrix (not hardcoded alliances)."""
    coalition          = set(r.get("coalition_members", []))
    relationship_matrix = r.get("relationship_matrix", {})  # {agent: {agent: float[-1,1]}}
    if len(coalition) < 2:
        return 0.0
    size_score = min(len(coalition) / 6.0, 1.0)
    pairs      = list(itertools.combinations(coalition, 2))
    # avg trust from live relationship matrix, normalized [-1,1]→[0,1]
    trust_vals = []
    for a, b in pairs:
        trust = relationship_matrix.get(a, {}).get(b,
                relationship_matrix.get(b, {}).get(a, 0.0))
        trust_vals.append((trust + 1.0) / 2.0)
    trust_coherence = sum(trust_vals) / len(trust_vals)
    # Durability: fraction consistent with prior-round stances
    durability      = r.get("coalition_durability", 0.7)   # from orchestrator tracking
    return size_score * trust_coherence * durability        # [0, 1]


def _economic_score(r: dict) -> float:
    """E: welfare-based resilience, penalizes shocks and sanctions costs."""
    growth            = r.get("gdp_growth_rate", 0.0)       # fractional
    inflation_shock   = r.get("inflation_shock", 0.0)       # [0,1]
    trade_disruption  = r.get("trade_disruption", 0.0)      # [0,1]
    sanctions_cost    = r.get("sanctions_cost", 0.0)        # [0,1]
    raw = growth - inflation_shock - trade_disruption * 0.5 - sanctions_cost * 0.3
    return max(0.0, min(1.0, (raw + 1.0) / 2.0))


def _humanitarian_score(r: dict) -> float:
    """H: civilian protection + law compliance (UNESCO mandate adherence)."""
    civilian_harm     = r.get("civilian_harm_index", 0.0)   # [0,1], higher=worse
    refugee_risk      = r.get("refugee_displacement_risk", 0.0)
    law_compliance    = r.get("law_compliance_score", 0.5)  # UNESCO citation rate
    raw = -civilian_harm - refugee_risk * 0.7 + law_compliance * 1.3
    return max(0.0, min(1.0, (raw + 1.7) / 3.0))


def _compute_penalties(r: dict) -> float:
    """Sum of hard constraint violations. Can be catastrophic (nuclear = -1.0)."""
    total = 0.0
    violations = r.get("constraint_violations", [])
    for v in violations:
        total += HARD_PENALTIES.get(v, -0.1)
    return total


def _future_value(r: dict) -> float:
    """V(s'): heuristic estimate of next-state stability."""
    rel_matrix = r.get("next_relationship_matrix", r.get("relationship_matrix", {}))
    crisis_resolved = r.get("crisis_resolved", False)
    if not rel_matrix:
        return 0.5
    all_scores = [v for row in rel_matrix.values() for v in row.values()]
    avg_trust  = sum(all_scores) / max(len(all_scores), 1)
    trust_norm = (avg_trust + 1.0) / 2.0
    stability  = trust_norm * (1.2 if crisis_resolved else 0.8)
    return max(0.0, min(1.0, stability))


def _counterfactual_advantage(r: dict) -> float:
    """A = Outcome(action) - Outcome(null_action). Rewards improvement over baseline."""
    # Null action = abstain, no policy intervention
    baseline_stability = r.get("null_action_stability", r.get("prev_stability", 0.5))
    current_stability  = r.get("current_stability", 0.5)
    return current_stability - baseline_stability   # can be negative (action made things worse)


def _robustness_reward(r: dict) -> float:
    """R_robust: performance under shock perturbation (optional, defaults to 0)."""
    return r.get("shock_robustness_score", 0.0)


class MOGSRGrader:
    """
    MOGSR — Multi-Objective Geopolitical Stability Reward.
    4-layer: immediate + long-horizon + counterfactual + robustness.
    Crisis-adaptive weights. Hard constraint penalties.
    """

    def score(self, round_result: dict, crisis_type: str = "DEFAULT") -> float:
        w = CRISIS_WEIGHTS.get(crisis_type, CRISIS_WEIGHTS["DEFAULT"])

        # Layer 1: Immediate multi-objective reward
        S = _security_score(round_result)
        D = _diplomacy_score(round_result)
        C = _coalition_score(round_result)
        E = _economic_score(round_result)
        H = _humanitarian_score(round_result)
        P = _compute_penalties(round_result)

        R_immediate = w["S"]*S + w["D"]*D + w["C"]*C + w["E"]*E + w["H"]*H + P

        # Layer 2: Long-horizon strategic value
        R_longterm  = GAMMA * _future_value(round_result)

        # Layer 3: Counterfactual advantage (reward improvement over no action)
        A           = _counterfactual_advantage(round_result)

        # Layer 4: Shock robustness bonus
        R_robust    = _robustness_reward(round_result)

        R_final = R_immediate + R_longterm + LAMBDA * A + BETA * R_robust

        # Hard episode termination on catastrophic penalty
        if "nuclear_escalation" in round_result.get("constraint_violations", []):
            return max(-1.0, R_immediate + P)   # override — catastrophic

        return max(-1.0, min(2.0, R_final))   # bounded for training stability


# ─── Per-task thin wrappers (required by openenv.yaml task.grader field) ─────

class CrisisResolutionGrader(MOGSRGrader):
    """Task: resolve a crisis. Uses all 5 objectives."""
    def score(self, r: dict) -> float:
        return super().score(r, crisis_type=r.get("crisis_type", "DEFAULT"))

class CoalitionGrader(MOGSRGrader):
    """Task: form a coalition. Upweights C and D; downweights E."""
    def score(self, r: dict) -> float:
        r["crisis_type"] = r.get("crisis_type", "DEFAULT")
        w_override = CRISIS_WEIGHTS.get(r["crisis_type"], CRISIS_WEIGHTS["DEFAULT"]).copy()
        w_override["C"] = 0.40   # coalition task — C is primary signal
        w_override["D"] = 0.30
        # Reduce others proportionally
        reduction = (w_override["C"] + w_override["D"]) - (0.15 + 0.20)
        w_override["S"] -= reduction * 0.5
        w_override["E"] -= reduction * 0.3
        w_override["H"] -= reduction * 0.2
        import copy
        r2 = copy.deepcopy(r)
        CRISIS_WEIGHTS["_tmp"] = w_override
        result = super().score(r2, crisis_type="_tmp")
        del CRISIS_WEIGHTS["_tmp"]
        return result

class DiplomacyGrader(MOGSRGrader):
    """Task: persuade opponents. Upweights D; uses stance_change signal."""
    def score(self, r: dict) -> float:
        # Inject stance efficiency into diplomacy score override
        initial_opp = r.get("initial_opposition_count", 0)
        if initial_opp == 0:
            r.setdefault("resolution_success", True)
        else:
            stance_changes = r.get("stance_changes", 0)
            r["resolution_success"] = stance_changes / initial_opp >= 0.5
        return super().score(r, crisis_type=r.get("crisis_type", "DEFAULT"))
```

#### New file: `inference.py` (required name, root dir)

**4-stage pipeline** (mirrors DisasterMan pattern judges know):
- Stage 1: PyTorch StabilityScorer — ranks country risk, identifies nuclear/conflict escalation signals
- Stage 2: Triage Agent (LLM) — assesses crisis severity, identifies which agents need to act
- Stage 3: Planner Agent (LLM) — 2-step lookahead for coalition/resolution strategy
- Stage 4: Action Agent (LLM) — generates final action + hard-constraint validation + fallback

**Model**: `meta-llama/Llama-3.1-8B-Instruct` via HF Serverless Inference API (strong, Meta-native, within $30 budget)

```python
import os, json, uuid, requests, math, torch
from datetime import datetime, timezone
from openai import OpenAI
from pytorch_scorer import StabilityScorer, score_stability

# ── Required env vars ─────────────────────────────────────────────────────────
API_BASE_URL = os.environ["API_BASE_URL"]   # HF Serverless: "https://api-inference.huggingface.co/v1"
MODEL_NAME   = os.environ["MODEL_NAME"]     # "meta-llama/Llama-3.1-8B-Instruct"
HF_TOKEN     = os.environ["HF_TOKEN"]
ENV_URL      = os.environ.get("ENV_URL", "http://localhost:7860")

client = OpenAI(base_url=API_BASE_URL, api_key=HF_TOKEN)

VALID_ACTION_TYPES = {"propose_resolution", "form_coalition", "veto", "abstain",
                      "invoke_article", "sanction"}

# ── Stage 1: PyTorch stability scoring ────────────────────────────────────────
def stage1_risk_analysis(obs: dict) -> dict:
    """Score each country's risk level using StabilityScorer. Sub-millisecond."""
    stability = score_stability(obs.get("country_pnl", {}), obs.get("relationship_matrix", {}))
    crisis_severity = obs.get("current_crisis", {}).get("severity", 0.5)
    nuclear_risk = any(
        obs.get("relationship_matrix", {}).get("DPRK", {}).get(a, 0) < -0.5
        for a in ["USA", "IND", "SAU"]
    )
    return {
        "stability_score": stability,
        "crisis_severity": crisis_severity,
        "nuclear_risk":    nuclear_risk,
        "priority_agents": ["DPRK", "RUS"] if nuclear_risk else ["UNESCO", "IND"],
    }

# ── Stage 2: Triage Agent ────────────────────────────────────────────────────
def stage2_triage(obs: dict, risk: dict, agent_id: str) -> str:
    """LLM assesses crisis and recommended stance."""
    prompt = f"""Crisis: {obs.get("current_crisis", {}).get("headline", "Global crisis")}
Stability score: {risk["stability_score"]:.2f}
Nuclear risk: {risk["nuclear_risk"]}
You represent: {agent_id}
In 1 sentence, what is the biggest risk right now and what should {agent_id} prioritize?"""
    resp = client.chat.completions.create(
        model=MODEL_NAME,
        messages=[{"role": "user", "content": prompt}],
        max_tokens=80, temperature=0.3,
    )
    return resp.choices[0].message.content.strip()

# ── Stage 3: Planner Agent ────────────────────────────────────────────────────
def stage3_plan(obs: dict, triage: str, agent_id: str) -> str:
    """LLM 2-step lookahead: what action leads to best outcome next round?"""
    rels = obs.get("relationship_matrix", {}).get(agent_id, {})
    allies = [a for a, v in rels.items() if v > 0.3]
    prompt = f"""Triage: {triage}
{agent_id}'s current allies (relationship > 0.3): {allies}
What single action should {agent_id} take? Choose from: propose_resolution, form_coalition,
veto, abstain, invoke_article, sanction. Consider 2 steps ahead. 1 sentence only."""
    resp = client.chat.completions.create(
        model=MODEL_NAME,
        messages=[{"role": "user", "content": prompt}],
        max_tokens=60, temperature=0.2,
    )
    return resp.choices[0].message.content.strip()

# ── Stage 4: Action Agent ────────────────────────────────────────────────────
def stage4_action(obs: dict, plan: str, agent_id: str) -> dict:
    """Generate structured action JSON. Hard constraint validation + fallback."""
    prompt = f"""Plan: {plan}
You are {agent_id}. Generate a diplomatic action as JSON only:
{{"action_type": "<one of: propose_resolution|form_coalition|veto|abstain|invoke_article|sanction>",
 "target": "<agent_id or null>",
 "description": "<policy justification, 1-2 sentences>"}}"""
    resp = client.chat.completions.create(
        model=MODEL_NAME,
        messages=[{"role": "user", "content": prompt}],
        max_tokens=150, temperature=0.1,
    )
    raw = resp.choices[0].message.content.strip()
    try:
        # Strip markdown fences if present
        clean = raw.replace("```json", "").replace("```", "").strip()
        data = json.loads(clean)
        if data.get("action_type") not in VALID_ACTION_TYPES:
            raise ValueError(f"Invalid action_type: {data.get('action_type')}")
        return {"agent_id": agent_id, **data}
    except Exception:
        # Deterministic fallback — never hallucinate an invalid action
        return {"agent_id": agent_id, "action_type": "abstain",
                "target": None, "description": f"{agent_id} observes and withholds position."}

# ── Main inference loop ────────────────────────────────────────────────────────
TASKS_TO_RUN = ["task_1", "task_2", "task_3"]   # run all 3 for grader coverage

for ep, task in enumerate(TASKS_TO_RUN, 1):
    session_id = str(uuid.uuid4())
    
    r = requests.post(f"{ENV_URL}/reset", json={"task": task, "session_id": session_id}, timeout=15)
    obs = r.json()["observation"]
    max_steps = r.json().get("max_steps", 5)
    
    ts = datetime.now(timezone.utc).isoformat()
    print(f'[START] {json.dumps({"episode": ep, "task": task, "model": MODEL_NAME, "timestamp": ts})}')
    
    total_reward = 0.0
    rounds       = []
    for step_num in range(1, max_steps + 1):
        agent_id = obs.get("active_agent", "USA")
        
        # 4-stage pipeline
        risk   = stage1_risk_analysis(obs)
        triage = stage2_triage(obs, risk, agent_id)
        plan   = stage3_plan(obs, triage, agent_id)
        action = stage4_action(obs, plan, agent_id)
        
        resp     = requests.post(f"{ENV_URL}/step",
                                  json={"action": action, "session_id": session_id}, timeout=30)
        data     = resp.json()
        obs, reward, done = data["observation"], data["reward"], data["done"]
        total_reward += reward
        rounds.append(data.get("round_result", {}))
        
        print(f'[STEP] {json.dumps({"step": step_num, "action": action, "reward": round(reward, 4), "state": obs})}')
        if done:
            break
    
    # Grade the episode
    grade_resp = requests.post(f"{ENV_URL}/grader",
                                json={"session_id": session_id, "task": task, "rounds": rounds},
                                timeout=10)
    grade = grade_resp.json() if grade_resp.ok else {}
    normalized = grade.get("normalized", (math.tanh(total_reward / max_steps * 2) + 1) / 2)
    
    print(f'[END] {json.dumps({"episode": ep, "task": task, "total_reward": round(total_reward, 4), "normalized": round(normalized, 4), "steps": step_num, "success": done})}')
```

**Reward hacking prevention (required per guide)**:
1. Cap description length in action — agents can't exploit long descriptions for reward
2. Validate action_type against allowlist — no invalid action types get positive reward
3. Check that coalition stances are consistent across rounds (durability check)
4. Hard constraint violations (nuclear escalation) terminate episode immediately — no reward gaming
5. DisasterMan's formula: `(tanh(cumul / max_steps * 2) + 1) / 2` — compresses extreme values

**SFT warm-up** (optional but improves GRPO convergence):
Before GRPO, run 50 steps of SFT on 500 generated "gold" debates (canned debates from orchestrator
where vote_passed=True). This ensures the model has non-zero probability of generating valid
action JSON before RL starts — prevents the "success probability = 0" failure mode.

# Cell 3: Prompt dataset — 200 debate scenario seeds
import json, requests, uuid
from datasets import Dataset

def generate_episode_seeds(n: int = 200) -> list[dict]:
    """Generate episode seed prompts for GRPO to sample from."""
    crisis_types = ["natural_disaster", "arms_race", "trade_war", 
                    "military_escalation", "war_outbreak", "cultural_destruction"]
    agent_ids    = ["USA", "CHN", "RUS", "IND", "DPRK", "SAU"]
    seeds = []
    for i in range(n):
        crisis = crisis_types[i % len(crisis_types)]
        agent  = agent_ids[i % len(agent_ids)]
        seeds.append({
            "prompt": f"You are {agent}. Crisis: {crisis}. Propose your diplomatic action as JSON.",
            "crisis": crisis,
            "agent_id": agent,
        })
    return seeds

dataset = Dataset.from_list(generate_episode_seeds(200))

# Cell 4: Reward function — calls the live env
def reward_fn(completions: list[str], prompts: list[str], **kwargs) -> list[float]:
    rewards = []
    for completion, prompt in zip(completions, prompts):
        try:
            # Parse prompt to extract agent_id
            agent_id = "USA"
            for agent in ["USA", "CHN", "RUS", "IND", "DPRK", "SAU"]:
                if f"You are {agent}" in prompt:
                    agent_id = agent
                    break
            
            session_id = str(uuid.uuid4())
            requests.post(f"{ENV_URL}/reset", json={"task": "crisis_resolution", 
                                                      "session_id": session_id}, timeout=10)
            action = {"agent_id": agent_id, "action_type": "propose_resolution",
                      "target": None, "description": completion[:300]}
            resp = requests.post(f"{ENV_URL}/step", 
                                  json={"action": action, "session_id": session_id}, timeout=10)
            rewards.append(float(resp.json().get("reward", 0.0)))
        except Exception:
            rewards.append(0.0)
    return rewards

# Cell 5: Train
from trl import GRPOTrainer, GRPOConfig
from transformers import AutoModelForCausalLM, AutoTokenizer

from transformers import BitsAndBytesConfig
bnb_config = BitsAndBytesConfig(load_in_4bit=True, bnb_4bit_compute_dtype="bfloat16")
model = AutoModelForCausalLM.from_pretrained(MODEL, quantization_config=bnb_config,
                                              device_map="auto")
tokenizer = AutoTokenizer.from_pretrained(MODEL)

config = GRPOConfig(
    num_generations=4,
    max_steps=200,           # ~30 min on T4 with env latency
    learning_rate=5e-6,
    gradient_checkpointing=True,
    output_dir="./worldpolicy-trained",
    logging_steps=10,
)

trainer = GRPOTrainer(
    model=model,
    tokenizer=tokenizer,
    reward_funcs=reward_fn,
    args=config,
    train_dataset=dataset,
)
trainer.train()

# Cell 6: Plot reward curve
import matplotlib.pyplot as plt
import pandas as pd

logs = pd.DataFrame(trainer.state.log_history)
plt.figure(figsize=(10, 4))
plt.plot(logs["step"], logs["reward"])
plt.xlabel("Training Step")
plt.ylabel("Mean Episode Reward")
plt.title("WorldPolicy-Env GRPO Training — Reward Curve")
plt.tight_layout()
plt.savefig("training_results/reward_curve.png", dpi=150)
